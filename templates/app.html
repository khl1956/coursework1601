@app.route('/login', methods=['GET', 'POST'])
def login():
    if request.method == 'POST':
        user = User.query.filter_by(user_email=request.form['email']).first()
        if user:
            try:
                assert user.user_password == request.form['password']
            except AssertionError:
                return render_template('login.html', message='Password is wrong!')
            else:
                session['username'] = request.form['email']
                return redirect('/')
        else:
            return render_template('login.html', message='User does not exist!')
    return render_template('login.html', message=' ')


@app.route('/unlogin', methods=['GET'])
def unlogin():
    session.pop('username', None)
    return redirect(url_for('login'))


@app.route('/signup', methods=['GET', 'POST'])
def signup():
    if request.method == 'POST':
        username = request.form['username']
        email = request.form['email']
        password = request.form['password']
        try:
            user = User(user_email=email, user_username=username, user_password=password)
            db.session.add(user)
            db.session.commit()
        except Exception as e:
            print(e)
            return render_template('login.html',
                                   message=f'Try to log in, because user with username {username} does exist.')
        return render_template('login.html', message=False)
    return render_template('signup.html')


def allowed_file(filename):
    print(filename)
    return '.' in filename and \
           filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS


@app.route('/upload', methods=['GET', 'POST'])
def upload():
    if session.get('username'):
        if request.method == 'POST':
            # check if the post request has the file part
            if 'file' not in request.files:
                return redirect(request.url)
            file = request.files['file']
            # if user does not select file, browser also
            # submit an empty part without filename
            if file.filename == '':
                return redirect(request.url)
            if file and allowed_file(file.filename):
                try:
                    path = os.path.join(app.config['UPLOAD_FOLDER'], 'app.py')
                    file.save(path)
                    file = Project(project_name=file.filename, project_link=path, user_email=session.get('username'))
                    db.session.add(file)
                    db.session.commit()
                except Exception as e:
                    print(e)
                    return render_template('error.html', message='Oops. Something is wrong. Try again.')
                else:
                    libpath = app.config['UPLOAD_FOLDER']
                    pdoc.import_path = [libpath]
                    print(pdoc.import_path)
                    mod = pdoc.import_module('app')
                    doc = pdoc.Module(mod)
                    string = doc.html()

                    path = os.path.join(os.path.abspath('templates'), 'app.html')

                    with open(path, "w+") as f:
                        f.write(string)

                    return render_template('app.html')

        return render_template('upload.html')
    return render_template('notlogin.html')


def detect_functions():
    import file as module
    functions = []
    objects = inspect.getmembers(module, inspect.isfunction)
    for object in objects:
        function, _ = object
        functions.append(function)
    return len(functions), functions
